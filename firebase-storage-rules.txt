// Firebase Console -> Storage -> Rules tab
// Paste this and click Publish.

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {

    // Chapter HTML files — uploaded by Host, read by all users
    match /chapters/{filename} {
      allow read: if true;
      allow write: if true;
    }

    // Quiz pool question images — uploaded via Host Panel quiz pool manager
    match /quiz_images/{filename} {
      allow read: if true;
      allow write: if true;
    }

    // Image Quiz photos — uploaded by Host for Time Attack mode
    match /imageQuizzes/{filename} {
      allow read: if true;
      allow write: if true;
    }

    // Reviewer Tool — pool JSON files + per-question images
    //   /quizReviews/_pools/v{X.Y}.json
    //   /quizReviews/v{X.Y}/{qid}.jpg
    match /quizReviews/{allPaths=**} {
      allow read: if true;
      allow write: if true;
    }
  }
}
