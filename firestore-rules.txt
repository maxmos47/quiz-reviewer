// Firebase Console -> Firestore Database -> Rules tab

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // quiz_pool — read by Web App (onSnapshot), written by Reviewer Tool + Host Panel
    match /quiz_pool/{docId} {
      allow read: if true;
      allow write: if true;
    }
    // quizReviews — Reviewer Tool workspace (real-time review patches + pool meta + audit_log)
    match /quizReviews/{document=**} {
      allow read, write: if true;
    }
    // All other collections (users, chapters, quizzes, config, imageQuizzes, etc.)
    match /{document=**} {
      allow read, write: if true;
    }
  }
}
